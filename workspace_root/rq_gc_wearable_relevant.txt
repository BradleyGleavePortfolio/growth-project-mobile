    41	// D-011 RQ-GC surgical sweep: track the QueryClients minted per test so we can
    42	// tear them down in afterEach. A finite gcTime (the previous 60_000) leaves a
    43	// real GC timer pending for a full minute after the suite finishes, which is
    44	// the open handle behind the "Jest did not exit one second after the test run
    45	// has completed" warning. gcTime: Infinity is never scheduled as a timer, so
    46	// the per-metric preference cache entry still survives the synchronous
    47	// getQueryData reads below (it is only collected on an explicit clear) while
    48	// nothing remains pending after the run.
    49	const mintedClients: QueryClient[] = [];
    50	
    51	function makeWrapper() {
    52	  const qc = new QueryClient({
    53	    defaultOptions: {
    54	      // gcTime must be non-zero here: we read the per-metric preference cache
    55	      // entry directly via getQueryData and it has no React Query observer, so
    56	      // a gcTime of 0 would garbage-collect it the instant it's written and
    57	      // the assertions below would race against the sweep. Infinity keeps the
    58	      // entry alive for the test AND schedules no GC timer (no leak).
    59	      queries: { retry: false, gcTime: Infinity, staleTime: 0 },
    60	      // The mutation cache GC is the real open handle here: every settled
    61	      // mutation schedules a removal timer (RQ default 5 min) that keeps the
    62	      // Node event loop alive long after the test finishes — this is what
    63	      // produced the "Jest did not exit" warning. Infinity is never scheduled.
    64	      mutations: { retry: false, gcTime: Infinity },
    65	    },
    66	  });
    67	  mintedClients.push(qc);
    68	  const Wrapper = ({ children }: { children: React.ReactNode }) => (
    69	    <QueryClientProvider client={qc}>{children}</QueryClientProvider>
    70	  );
    71	  return { qc, Wrapper };
    72	}
