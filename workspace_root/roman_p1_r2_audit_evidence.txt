    36	 * time Roman has been opened (R1 UX finding U1). The four registers are taken
    37	 * verbatim from the identity spec:

    54	  const name = (input.firstName ?? '').trim();
    55

    58	    return 'Good day. My name is Roman. I will be looking after things here. Whenever you need me, I am present.';
    59	  }
    60	
    61	  if (input.surface === 'coach') {
    62	    // identity spec §2.3 coach operational register.
    63	    if (name === '') {
    64	      return 'I am ready. What needs attention?';
    65	    }
    66	    return `Good morning, ${name}. I am ready. What needs attention?`;
    67	  }

    70	  if (name === '') {
    71	    // Nameless ambient register — identity spec §2.1 greeting clause.
    72	    return 'Good day. Everything is in order. Where shall we begin?';
    73	  }
    74	  return `Welcome back, ${name}. Everything is in order. Where shall we begin?`;
