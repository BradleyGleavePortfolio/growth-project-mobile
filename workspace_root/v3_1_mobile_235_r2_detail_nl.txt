     1	/**
     2	 * CommunityChallengeDetailScreen — the full-screen detail for one community
     3	 * challenge (v3-1). Route param: `challengeId`.
     4	 *
     5	 * BEHAVIORAL DESIGN (DESIGN_INTELLIGENCE Part III + §5.1 Screen Design Protocol):
     6	 *   - The screen leads with the participant's OWN progress in real metric units
     7	 *     — a competence signal (§3.7), never a "behind"/"losing" framing (§3.4).
     8	 *   - The cohort-LOCAL leaderboard (Strava §3.2) is STRICTLY OPT-IN: it is shown
     9	 *     only when the coach enabled it for the challenge AND the caller has opted
    10	 *     in. Until then the surface offers a calm, no-pressure opt-in affordance and
    11	 *     never reveals anyone's standing — zero shame, zero public failure (§3.4).
    12	 *   - One primary action (Hick's Law): Join when not joined, otherwise Log
    13	 *     progress via the ChallengeProgressSheet. Encouragement comments are a
    14	 *     secondary, social-proof surface, reportable for moderation.
    15	 *
    16	 * FLAG POSTURE: this route is only registered in CommunityNavigator when
    17	 * `featureFlags.communityChallenges` is true, so when the flag is OFF the screen
    18	 * never enters the tree. As a defense-in-depth guard the body still renders a
    19	 * neutral "not available" state if it is somehow reached with the flag off.
    20	 *
    21	 * DATA: uses `communityChallengesApi` directly via react-query (the shared
    22	 * `useCommunity` hooks are out of this slice's scope). Every mutation invalidates
    23	 * the relevant query so the UI reflects the server's monotonic truth. Tokens
    24	 * only (no raw hex); line Ionicons only (no emoji); reduced-motion aware via the
    25	 * progress sheet; real loading / empty / error states (no spinner-only screens).
    26	 */

    62	/** A human, non-shaming reason for an error surface (no raw error leakage). */
    63	function describeError(err: unknown): string {
    64	  if (err instanceof CommunityApiError) {
    65	    switch (err.kind) {
    66	      case 'forbidden':
    67	        return 'This challenge belongs to a cohort you are not part of.';
    68	      case 'gone':
    69	        return 'This challenge is no longer available.';
    70	      case 'conflict':
    71	        return 'Your progress was updated elsewhere. We have refreshed it for you.';
    72	      case 'network':
    73	        return 'We could not reach the server. Check your connection and try again.';
    74	      case 'contract':
    75	        return 'Something looks off on our end. Please try again shortly.';
    76	      default:
    77	        return 'We could not load this challenge. Please try again.';
    78	    }
    79	  }
    80	  return 'We could not load this challenge. Please try again.';
    81	}
