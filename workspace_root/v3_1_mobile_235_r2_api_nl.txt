     3	 * backend.
     4	 *
     5	 * Backend contract source of truth (binding — do NOT drift):
     6	 *   growth-project-backend/src/community/challenges/community-challenges.controller.ts
     7	 *   growth-project-backend/src/community/challenges/community-challenges.dto.ts
     8	 *
     9	 * Wire posture mirrors communityApi.ts:
    10	 *   - Every response is validated with Zod at the boundary so a drifted shape
    11	 *     THROWS here (wrapped as a `contract` error) instead of feeding malformed
    12	 *     data into React state.
    13	 *   - Mutations send an Idempotency-Key (R19) so a double-tap deduplicates.
    14	 *   - Behavioral framing (design Part III): the leaderboard is OFF unless the
    15	 *     coach enabled it AND the caller opted in; the API never returns a
    16	 *     non-consenting participant's row and never a "you are losing" signal.
    17	 */

   151	function classify(status: number): CommunityApiError['kind'] {
   152	  if (status === 401) return 'unauthorized';
   153	  if (status === 403) return 'forbidden';
   154	  if (status === 409) return 'conflict';
   155	  if (status === 410) return 'gone';
   156	  if (status >= 500) return 'server';
   157	  if (status === 0) return 'network';
   158	  return 'unknown';
   159	}
   160	
   161	async function call<T>(
   162	  schema: z.ZodType<T>,
   163	  fn: () => Promise<{ data: unknown }>,
   164	): Promise<T> {
   165	  let res: { data: unknown };
   166	  try {
   167	    res = await fn();
   168	  } catch (err) {
   169	    if (axios.isAxiosError(err)) {
   170	      const status = err.response?.status ?? 0;
   171	      throw new CommunityApiError(
   172	        classify(status),
   173	        status,
   174	        `community challenge request failed (${status || 'network'})`,
   175	        err,
   176	      );
   177	    }
   178	    throw new CommunityApiError(
   179	      'unknown',
   180	      -1,
   181	      'community challenge request failed',
   182	      err,
   183	    );
   184	  }
   185	  try {
   186	    return schema.parse(res.data);
   187	  } catch (err) {
   188	    if (err instanceof z.ZodError) {
   189	      throw new CommunityApiError(
   190	        'contract',
   191	        200,
   192	        'challenge response shape drifted from the backend contract',
   193	        err,
   194	      );
   195	    }
   196	    throw err;
   197	  }
   198	}

   200	function idempotentHeaders(): { headers: Record<string, string> } {
   201	  return { headers: { 'Idempotency-Key': generateIdempotencyKey() } };
   202	}

   206	export const communityChallengesApi = {
   207	  /** GET /community/workspaces/:workspaceId/challenges */
   208	  listChallenges(
   209	    workspaceId: string,
   210	    opts: { cohortId?: string; status?: CommunityChallengeStatus } = {},
   211	  ): Promise<CommunityChallenge[]> {
   212	    const params: Record<string, string> = {};
   213	    if (opts.cohortId) params.cohort_id = opts.cohortId;
   214	    if (opts.status) params.status = opts.status;
   215	    return call(ChallengeListResponseSchema, () =>
   216	      api.get<unknown>(`/community/workspaces/${workspaceId}/challenges`, {
   217	        params,
   218	      }),
   219	    ).then((r) => r.challenges);
   220	  },
   221	
   222	  /** GET /community/challenges/:challengeId — definition + caller's own row. */
   223	  getChallenge(challengeId: string): Promise<CommunityChallengeDetail> {
   224	    return call(ChallengeResponseSchema, () =>
   225	      api.get<unknown>(`/community/challenges/${challengeId}`),
   226	    );
   227	  },
   228	
   229	  /** POST /community/challenges/:challengeId/join — idempotent join. */
   230	  join(challengeId: string): Promise<CommunityChallengeParticipation> {
   231	    return call(ParticipationResponseSchema, () =>
   232	      api.post<unknown>(
   233	        `/community/challenges/${challengeId}/join`,
   234	        {},
   235	        idempotentHeaders(),
   236	      ),
   237	    ).then((r) => r.participation);
   238	  },
   239	
   240	  /** PUT /community/challenges/:challengeId/progress — log progress (monotonic). */
   241	  updateProgress(
   242	    challengeId: string,
   243	    progressValue: number,
   244	  ): Promise<CommunityChallengeParticipation> {
   245	    return call(ParticipationResponseSchema, () =>
   246	      api.put<unknown>(
   247	        `/community/challenges/${challengeId}/progress`,
   248	        { progress_value: progressValue },
   249	        idempotentHeaders(),
   250	      ),
   251	    ).then((r) => r.participation);
   252	  },
   253	
   254	  /** PUT /community/challenges/:challengeId/leaderboard-opt-in — opt in/out. */
   255	  setLeaderboardOptIn(
   256	    challengeId: string,
   257	    optedIn: boolean,
   258	  ): Promise<CommunityChallengeParticipation> {
   259	    return call(ParticipationResponseSchema, () =>
   260	      api.put<unknown>(
   261	        `/community/challenges/${challengeId}/leaderboard-opt-in`,
   262	        { opted_in: optedIn },
   263	        idempotentHeaders(),
   264	      ),
   265	    ).then((r) => r.participation);
   266	  },
   267	
   268	  /** GET /community/challenges/:challengeId/leaderboard — opt-in gated. */
   269	  getLeaderboard(challengeId: string): Promise<CommunityChallengeLeaderboard> {
   270	    return call(LeaderboardResponseSchema, () =>
   271	      api.get<unknown>(`/community/challenges/${challengeId}/leaderboard`),
   272	    );
   273	  },
   274	
   275	  /** GET /community/challenges/:challengeId/comments */
   276	  listComments(challengeId: string): Promise<CommunityChallengeComment[]> {
   277	    return call(ChallengeCommentListResponseSchema, () =>
   278	      api.get<unknown>(`/community/challenges/${challengeId}/comments`),
   279	    ).then((r) => r.comments);
   280	  },
   281	
   282	  /** POST /community/challenges/:challengeId/comments — encouragement. */
   283	  addComment(
   284	    challengeId: string,
   285	    body: string,
   286	  ): Promise<CommunityChallengeComment> {
   287	    return call(ChallengeCommentResponseSchema, () =>
   288	      api.post<unknown>(
   289	        `/community/challenges/${challengeId}/comments`,
   290	        { body },
   291	        idempotentHeaders(),
   292	      ),
   293	    ).then((r) => r.comment);
   294	  },
   295	
   296	  /** POST /community/challenges/:challengeId/comments/:commentId/report */
   297	  reportComment(
   298	    challengeId: string,
   299	    commentId: string,
   300	    reason: string,
   301	    notes?: string,
   302	  ): Promise<void> {
   303	    return call(z.unknown(), () =>
   304	      api.post<unknown>(
   305	        `/community/challenges/${challengeId}/comments/${commentId}/report`,
   306	        { target_type: 'comment', target_id: commentId, reason, notes },
   307	        idempotentHeaders(),
   308	      ),
   309	    ).then(() => undefined);
   310	  },
   311	};

   313	export type CommunityChallengesApi = typeof communityChallengesApi;
