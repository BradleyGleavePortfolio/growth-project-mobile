--- ChallengeProgressSheet
     1	/**
     2	 * ChallengeProgressSheet — the bottom sheet where a participant logs progress
     3	 * toward their OWN target for a community challenge (v3-1).
     4	 *
     5	 * BEHAVIORAL DESIGN (DESIGN_INTELLIGENCE Part III + §5.1):
     6	 *   - Emotional target: the participant leaves feeling CAPABLE — they see their
     7	 *     own progress grow in real metric units (competence feedback §3.7), never
     8	 *     a comparison against anyone else here.
     9	 *   - Primary path (§5.1 Step 2): a single "Log progress" action. The current
    10	 *     value is pre-filled (smart default, Hick's Law) so the common case is one
    11	 *     adjustment + one tap.
    12	 *   - Progress is MONOTONIC and framed only as growth: the bar never shrinks and
    13	 *     there is no "behind"/"failed"/"you're losing" state (§3.4, no public
    14	 *     failure). Reaching the goal is a positive closure celebration (§5.1 Step 6).
    15	 *   - Reduced motion: the fill animation collapses to an instant set when the OS
    16	 *     "Reduce Motion" setting is on.
    17	 *
    18	 * Tokens only (no raw hex). Line Ionicons only (no emoji). Accessible (>=48dp
    19	 * targets, labelled controls, live region on the progress value).
    20	 */
    21	import React, { useCallback, useEffect, useRef, useState } from 'react';
    22	import {
    23	  AccessibilityInfo,
    24	  Animated,
    25	  Modal,
    26	  StyleSheet,
    27	  Text,
    28	  TextInput,
    29	  View,
    30	} from 'react-native';
    31	import * as Haptics from 'expo-haptics';
    32	import { Ionicons } from '@expo/vector-icons';
    33	import HapticPressable from '../HapticPressable';
    34	import { useTheme } from '../../theme/useTheme';
    35	import { spacing, radius, motion } from '../../theme/tokens';
    36	import type {
    37	  CommunityChallenge,
    38	  CommunityChallengeParticipation,
    39	} from '../../api/communityChallengesApi';
    40	
    41	export interface ChallengeProgressSheetProps {
    42	  visible: boolean;
    43	  challenge: CommunityChallenge;
    44	  /** The caller's participation, or null when they have not joined yet. */
    45	  participation: CommunityChallengeParticipation | null;
    46	  /**
    47	   * Persist a new cumulative progress value. Resolves with the server-confirmed
    48	   * outcome (whether the goal is now complete) so the sheet can stage the
    49	   * completion peak; REJECTS when the write fails so the sheet surfaces a calm
    50	   * error and keeps the user's draft (no silent swallow, FIFTY_FAILURES #36).
    51	   */
    52	  onSubmit: (progressValue: number) => Promise<{ completed: boolean }>;
    53	  onClose: () => void;
    54	  /** Submission in flight — disables the primary action and shows progress copy. */
    55	  submitting?: boolean;
    56	  testID?: string;
    57	}
    58	
    59	/** Clamp a fraction into [0, 1]; null target → null (no bar, only a raw value). */
    60	function fractionFor(value: number, target: number | null): number | null {
    61	  if (target === null || target <= 0) return null;
    62	  return Math.min(Math.max(value / target, 0), 1);
    63	}
    64	
    65	export default function ChallengeProgressSheet({
    66	  visible,
    67	  challenge,
    68	  participation,
    69	  onSubmit,
    70	  onClose,
    71	  submitting = false,
    72	  testID,
    73	}: ChallengeProgressSheetProps): React.ReactElement {
    74	  const { semanticColors } = useTheme();
    75	  const current = participation?.progress_value ?? 0;
    76	  const target = challenge.target_value;
    77	  const unit = challenge.unit ?? '';
    78	  const [draft, setDraft] = useState<string>(String(current));
    79	  const [reduceMotion, setReduceMotion] = useState(false);
    80	  // Calm, surfaced error for a rejected submit (no silent swallow).
    81	  const [submitError, setSubmitError] = useState<string | null>(null);
    82	  // Completion PEAK: after a server-confirmed goal completion the sheet stays
    83	  // open in this state (fill at 100%, success haptic, closure copy + Done).
    84	  const [celebrating, setCelebrating] = useState(false);
    85	  const fill = useRef(new Animated.Value(fractionFor(current, target) ?? 0)).current;
    86	
    87	  // Re-seed the draft to the current value whenever the sheet (re)opens — the
    88	  // smart default is "your number so far", so logging more is a small edit.
    89	  // Also clear any prior error / celebration when the sheet (re)opens.
    90	  useEffect(() => {
    91	    if (visible) {
    92	      setDraft(String(current));
    93	      setSubmitError(null);
    94	      setCelebrating(false);
    95	    }
    96	  }, [visible, current]);
    97	
    98	  useEffect(() => {
    99	    let mounted = true;
   100	    void AccessibilityInfo.isReduceMotionEnabled().then((on) => {
   101	      if (mounted) setReduceMotion(on);
   102	    });
   103	    const sub = AccessibilityInfo.addEventListener(
   104	      'reduceMotionChanged',
   105	      (on) => setReduceMotion(on),
   106	    );
   107	    return () => {
   108	      mounted = false;
   109	      sub.remove();
   110	    };
   111	  }, []);
   112	
   113	  const draftValue = Number(draft);
   114	  const draftValid = draft.trim() !== '' && Number.isFinite(draftValue) && draftValue >= 0;
   115	  // Monotonic: a log can only hold or raise the visible number.
   116	  const nextValue = draftValid ? Math.max(draftValue, current) : current;
   117	  // When the typed number is BELOW the saved total we silently keep the higher
   118	  // saved value (monotonic). We explain that inline as reassurance, NOT as an
   119	  // error (no red, no shame) — UX finding 11 / §3.4.
   120	  const isClampedDown = draftValid && current > 0 && draftValue < current;
   121	  const nextFraction = fractionFor(nextValue, target);
   122	  const willComplete =
   123	    target !== null && target > 0 && nextValue >= target && current < target;
   124	
   125	  // Animate the bar toward the DRAFT fraction so the user previews their gain
   126	  // before committing — anticipation, then the submit confirms it. During the
   127	  // completion PEAK the bar is driven to a full 100% instead (see below), so we
   128	  // skip the draft-preview animation while celebrating.
   129	  useEffect(() => {
   130	    if (celebrating) return;
   131	    const toValue = nextFraction ?? 0;
   132	    if (reduceMotion) {
   133	      fill.setValue(toValue);
   134	      return;
   135	    }
   136	    const anim = Animated.timing(fill, {
   137	      toValue,
   138	      duration: motion.duration.base,
   139	      useNativeDriver: false,
   140	    });
   141	    anim.start();
   142	    return () => anim.stop();
   143	  }, [nextFraction, reduceMotion, fill, celebrating]);
   144	
   145	  // Completion PEAK (UX finding 4 / §5.1 Step 6): once the server confirms the
   146	  // goal is reached we keep the sheet open and drive the fill to a full 100%.
   147	  // Reduced motion collapses the animation to an instant set; either way a
   148	  // single success haptic fires — calm, deliberate closure, no confetti.
   149	  useEffect(() => {
   150	    if (!celebrating) return;
   151	    if (reduceMotion) {
   152	      fill.setValue(1);
   153	    } else {
   154	      const anim = Animated.timing(fill, {
   155	        toValue: 1,
   156	        duration: motion.duration.base,
   157	        useNativeDriver: false,
   158	      });
   159	      anim.start();
   160	    }
   161	    void Haptics.notificationAsync(
   162	      Haptics.NotificationFeedbackType.Success,
   163	    ).catch(() => {
   164	      // Web / unsupported hardware — the visual closure still stands.
   165	    });
   166	  }, [celebrating, reduceMotion, fill]);
   167	
   168	  const handleSubmit = useCallback(async () => {
   169	    if (!draftValid || submitting) return;
   170	    setSubmitError(null);
   171	    try {
   172	      const result = await onSubmit(nextValue);
   173	      if (result.completed) {
   174	        // Stay open for the closure peak; the user dismisses with Done.
   175	        setCelebrating(true);
   176	      } else {
   177	        onClose();
   178	      }
   179	    } catch {
   180	      // Surface a calm, non-shaming error and KEEP the draft so the user can
   181	      // retry (no silent swallow, no optimistic state left dangling — #30/#36).
   182	      setSubmitError(
   183	        'We could not save your progress just now. Please try again.',
   184	      );
   185	    }
   186	  }, [draftValid, submitting, onSubmit, nextValue, onClose]);
   187	
   188	  const widthInterpolation = fill.interpolate({
   189	    inputRange: [0, 1],
   190	    outputRange: ['0%', '100%'],
   191	    extrapolate: 'clamp',
   192	  });
   193	
   194	  const percentLabel =
   195	    nextFraction === null ? null : `${Math.round(nextFraction * 100)}%`;
   196	  const goalLabel =
   197	    target === null
   198	      ? `${nextValue}${unit ? ` ${unit}` : ''} logged`
   199	      : `${nextValue} of ${target}${unit ? ` ${unit}` : ''}`;
   200	
   201	  return (
   202	    <Modal
   203	      visible={visible}
   204	      animationType={reduceMotion ? 'fade' : 'slide'}
   205	      transparent
   206	      onRequestClose={onClose}
