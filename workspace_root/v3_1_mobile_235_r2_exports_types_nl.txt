     1	/**
     2	 * communityNavTypes — typed param list + shared nav props for the Community
     3	 * sub-stack (v1-5). The stack is mounted by ClientNavigator ONLY when
     4	 * `featureFlags.communityTab` is true (see ClientNavigator.tsx). The deep-link
     5	 * route is likewise registered only behind that flag.
     6	 */
     7	import type {
     8	  NavigationProp,
     9	  RouteProp,
    10	  ParamListBase,
    11	} from '@react-navigation/native';
    12	
    13	export type CommunityStackParamList = {
    14	  /** Container with the Space sub-tab switcher (Today / Hall / Cohorts / DMs). */
    15	  CommunityTab: undefined;
    16	  /** The "today" object — universal home for what's happening for the client. */
    17	  CommunityToday: undefined;
    18	  /** A Space view: Hall (workspace) or a specific Cohort. */
    19	  CommunitySpace: { space: 'hall' | 'cohort'; cohortId?: string };
    20	  /** Single thread / post detail. */
    21	  CommunityThread: { postId: string };
    22	  /** DM inbox. */
    23	  CommunityDmList: undefined;
    24	  /** Single DM conversation. */
    25	  CommunityDmThread: { recipientId: string; participantLabel?: string };
    26	  /** Compose a post (or, with a recipient, a DM). */
    27	  CommunityComposer:
    28	    | { mode: 'post' }
    29	    | { mode: 'dm'; recipientId: string }
    30	    | undefined;
    31	  /** Single challenge detail (v3-1). Registered only behind communityChallenges. */
    32	  CommunityChallengeDetail: { challengeId: string };
    33	  /** Challenge discovery list (v3-1). Registered only behind communityChallenges. */
    34	  CommunityChallenges: undefined;
    35	};
    36	
    37	/** Loosely-typed nav prop used by the screens (matches the codebase pattern). */
    38	export type CommunityNav = NavigationProp<ParamListBase>;
    39	export type CommunityRoute<T extends keyof CommunityStackParamList> = RouteProp<
    40	  Record<string, CommunityStackParamList[T]>,
    41	  string
    42	>;
